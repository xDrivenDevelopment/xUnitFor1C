Процедура ЗакрытьФорму()

	Если _Форма.Открыта() Тогда
		_Форма.Модифицированность = Ложь;
		Попытка
			_Форма.Закрыть();
		Исключение

			НачатьТранзакцию();

			_Форма.УстановитьДействие("ПриЗакрытии", Неопределено);
			_Форма.Закрыть();
			ВызватьИсключение Ошибка;

	Иначе
		Попытка
			_Форма.Закрыть();
		Исключение
		КонецПопытки;
	КонецЕсли;
